public static void main(String[] args) {
        
        Scanner scanner = new Scanner(System.in);

        // Laço para cadastrar múltiplos clientes
        while (true) {
            Cliente cliente = new Cliente();
            System.out.println("Cadastro de Cliente");
            System.out.print("Nome: ");
            cliente.setNome(scanner.nextLine());
            System.out.print("CPF: ");
            cliente.setCpf(scanner.nextLine());
            System.out.print("Endereco: ");
            cliente.setEndereco(scanner.nextLine());
            System.out.print("Data de Nascimento: ");
            cliente.setDataNasc(scanner.nextLine());

            // Adicionando o cliente à lista
            ListaCliente.lista.add(cliente);

            // Verifica se deseja cadastrar outro cliente
            System.out.print("Deseja cadastrar outro cliente? (S/N): ");
            String resposta = scanner.nextLine();
            if (!resposta.equalsIgnoreCase("S")) {
                break; // Sai do laço se a resposta não for "S"
            }
        }

        // Imprime os clientes cadastrados
        System.out.println("\nClientes Cadastrados:");
        for (Cliente cliente : ListaCliente.listar()) {
            System.out.println("Nome: " + cliente.getNome());
            System.out.println("CPF: " + cliente.getCpf());
            System.out.println("Endereco: " + cliente.getEndereco());
            System.out.println("Data de Nascimento: " + cliente.getDataNasc());
            System.out.println();
        }
    }
}
